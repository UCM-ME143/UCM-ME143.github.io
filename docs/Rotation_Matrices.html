Rotation Matrix

 
% Let's review our rotation matrices
 
% Given a PSI rotation angle, the corresponding rotation matrix is given
 
psi = 10;
 
psi_in_radians = deg2rad(psi);
 
Rpsi = [ cos(psi_in_radians), sin(psi_in_radians), 0;
-sin(psi_in_radians), cos(psi_in_radians), 0;
0, 0, 1];
disp('Rotation around Psi = 10');
Rotation around Psi = 10
disp(Rpsi);
0.9848 0.1736 0 -0.1736 0.9848 0 0 0 1.0000
 
% We can do the same for the other angles theta and phi
 
theta = 10;
 
theta_in_radians = deg2rad(theta);
 
Rthe = [cos(theta_in_radians), 0, -sin(theta_in_radians);
0, 1, 0;
sin(theta_in_radians), 0, cos(theta_in_radians)];
 
disp('Rotation around Theta = 10');
Rotation around Theta = 10
disp(Rthe);
0.9848 0 -0.1736 0 1.0000 0 0.1736 0 0.9848
 
phi = 10;
 
phi_in_radians = deg2rad(phi);
 
Rphi = [ 1, 0, 0;
0, cos(phi_in_radians), sin(phi_in_radians);
0, -sin(phi_in_radians), cos(phi_in_radians)];
 
disp('Rotation around Phi = 10');
Rotation around Phi = 10
disp(Rphi);
1.0000 0 0 0 0.9848 0.1736 0 -0.1736 0.9848
 
% Now we can rotate a point by psi, theta and phi.
% Moving between the BODY and WORLD frames
 
p = [1;0;0];
% We set the point as a 3 row, 1 column matrix - each row corresponding to
% x, y and z coordinates.
 
 
% Lets do a rotation for p - let's assume that p is in the WORLD frame and
% we're going to move it into the BODY frame
 
new_point = Rphi*Rthe*Rpsi*p;
 
% Remember that the correct order is to rotate around psi first, then
% theta, then phi. In Matlab, we would write that with the first rotation,
% closest to point p.
 
disp('Starting at point p');
Starting at point p
disp(p);
1 0 0
disp('Rotated by Psi, Theta, Phi');
Rotated by Psi, Theta, Phi
disp(new_point);
0.9698 -0.1413 0.1986

Tower Example

% Let's do an example
 
% A tower is at 87 ft North and 30 ft West of our aircraft. Lets assume
% that the tower is 50 ft tall. Our airplane is taking off from the ground
% heading (psi) at -25 degrees and pitching up by 10 degrees. If the plane
% only goes forward, will the plane crash into the tower?
 
% We'll start by putting the coordinates of the tower (WORLD FRAME) into an
% array.
tower = [87;-30;-50];
 
% Next we'll calculate the rotation matrix. The angles represent the
% orientation of the aircraft.
 
psi = -25;
theta = 10;
phi = 0;
 
psi_in_radians = deg2rad(psi);
Rpsi = [ cos(psi_in_radians), sin(psi_in_radians), 0;
-sin(psi_in_radians), cos(psi_in_radians), 0;
0, 0, 1];
theta_in_radians = deg2rad(theta);
Rthe = [cos(theta_in_radians), 0, -sin(theta_in_radians);
0, 1, 0;
sin(theta_in_radians), 0, cos(theta_in_radians)];

0, cos(phi_in_radians), sin(phi_in_radians);

Using Functions

% It can get tiresome to keep having to copy-paste or rerun the rotation
% matrix equations every time. It's easier if we rewrite it as a
% programming function.
 
 
% To write a function, click on the arrow below NEW in Matlab, then select
% either 'Function' or 'Live Function' Use 'Function' for regular *.m
% files and use 'Live Function' for *.mlx files.
 
% In the new file, write in your function
 
function out = function_name(function_input1,function_input2)
out = function_input1 + function_input2;
end
 
% Now we can use it in our script
 
testing = function_name(1,2);
 
disp(testing);
3
 
% Lets make a function to generate a rotation matrix for Phi
 
function R = rotate_phi(angle)
angle = deg2rad(angle);
R = [1,0,0;
0,cos(angle),sin(angle);
0,-sin(angle),cos(angle)];
end
 
% Now we can do
 
Rphi = rotate_phi(50);
 
disp('Rotation Matrix - 50 degrees Phi');
Rotation Matrix - 50 degrees Phi
disp(Rphi);
1.0000 0 0 0 0.6428 0.7660 0 -0.7660 0.6428
 
% We can repeat that for the other rotations as well.
 
Rpsi = rotate_psi(10);
Rthe = rotate_theta(-30);
 
% And now we can calculate our WORLD to BODY rotation matrix
 
R_world_2_body = Rphi*Rthe*Rpsi;
 
disp('Rotation Matrix - WORLD to BODY');
Rotation Matrix - WORLD to BODY
disp(R_world_2_body);
0.8529 0.1504 0.5000 -0.4888 0.5665 0.6634 -0.1835 -0.8102 0.5567
 
% And if we want to go the other way, we take the transpose:
 
R_body_2_world = R_world_2_body';
disp('Rotation Matrix - BODY to WORLD');
Rotation Matrix - BODY to WORLD
disp(R_body_2_world);
0.8529 -0.4888 -0.1835 0.1504 0.5665 -0.8102 0.5000 0.6634 0.5567
 
 
% We can compact this a bit more
 
% We can create a function that calls other functions
 
function R = Rotate_World_2_Body(psi,theta,phi)
Rpsi = rotate_psi(psi);
Rthe = rotate_theta(theta);
Rphi = rotate_phi(phi);
R = Rphi*Rthe*Rpsi;
end
